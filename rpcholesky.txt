# R/rpcholesky.R

#' Accelerated Randomly Pivoted Cholesky Decomposition
#'
#' Computes a low-rank approximation of a positive semi-definite matrix using
#' the accelerated randomly pivoted Cholesky algorithm. This implementation is
#' based on the method described in "Randomly Pivoted Cholesky: Practical 
#' Approximation of a Kernel Matrix with Few Entry Evaluations" by Ethan N. Epperly.
#'
#' @param A Matrix or matrix-like object for low-rank approximation
#' @param k Target rank of the approximation
#' @param b Block size; defaults to k/10
#' @param stoptol Tolerance for early stopping (default = 1e-13)
#' @param verbose Whether to print progress information (default = FALSE)
#'
#' @return An object of class "rpchol" containing:
#'   \item{G}{The Gram factor}
#'   \item{idx}{Selected indices}
#'   \item{rows}{Selected rows}
#'   \item{k}{Target rank}
#'
#' @examples
#' # Create a random PSD matrix
#' set.seed(42)
#' X <- matrix(rnorm(100*10), 100, 10)
#' A <- X %*% t(X)
#' 
#' # Compute a rank-20 approximation
#' result <- rpcholesky(A, k = 20)
#' 
#' # Compute approximation error
#' rel_error <- approximation_error(A, result, relative = TRUE)
#' print(paste("Relative error:", rel_error))
#'
#' @export
rpcholesky <- function(A, k, b = NULL, stoptol = 1e-13, verbose = FALSE) {
  result <- accelerated_rpcholesky(A, k, b, stoptol, verbose)
  
  # Add k to the result
  result$k <- k
  
  # Set class to allow for custom methods
  class(result) <- "rpchol"
  
  return(result)
}

#' Print method for rpchol objects
#'
#' @param x An rpchol object
#' @param ... Additional arguments (not used)
#'
#' @export
print.rpchol <- function(x, ...) {
  cat("Accelerated Randomly Pivoted Cholesky Decomposition\n")
  cat("Rank:", x$k, "\n")
  cat("Selected", length(x$idx), "indices\n")
}

#' Create a Nystrom extension from an rpchol object
#'
#' Creates a Nystrom extension that can be used for efficient matrix operations.
#'
#' @param rpchol An rpchol object from rpcholesky()
#' @param A The original matrix (optional if already in rpchol)
#'
#' @return A nystrom object
#'
#' @export
create_nystrom <- function(rpchol, A = NULL) {
  if (is.null(A) && !is.null(rpchol$A)) {
    A <- rpchol$A
  }
  
  if (is.null(A)) {
    stop("Original matrix A is required to create Nystrom extension")
  }
  
  nys <- nystrom_extension(A, rpchol$idx)
  class(nys) <- "nystrom"
  
  return(nys)
}

#' Matrix multiplication for nystrom objects
#'
#' Multiplies a matrix by the Nystrom approximation.
#'
#' @param nys A nystrom object
#' @param B Matrix to multiply with
#'
#' @return The product of the Nystrom approximation and B
#'
#' @export
nystrom_multiply <- function(nys, B) {
  return(nystrom_mult(nys, B))
}

#' Compute the approximation error
#'
#' @param A The original matrix
#' @param approx An rpchol object or G matrix from rpcholesky
#' @param relative Whether to return the relative error (default = FALSE)
#'
#' @return The approximation error
#'
#' @export
approximation_error <- function(A, approx, relative = FALSE) {
  if (inherits(approx, "rpchol")) {
    G <- approx$G
  } else {
    G <- approx
  }
  
  return(approximation_error(A, G, relative))
}

#' Compute the trace of an rpchol object
#'
#' @param rpchol An rpchol object from rpcholesky()
#'
#' @return The trace of the low-rank approximation
#'
#' @export
trace_rpchol <- function(rpchol) {
  return(sum(rpchol$G^2))
}

#' Compute the matrix form of an rpchol object
#'
#' @param rpchol An rpchol object from rpcholesky()
#'
#' @return The matrix form of the low-rank approximation
#'
#' @export
as.matrix.rpchol <- function(rpchol) {
  return(t(rpchol$G) %*% rpchol$G)
}

#' Generate a test matrix with the "smile" pattern
#'
#' Creates a kernel matrix with the smile pattern used in the original paper.
#'
#' @param n Number of points
#' @param bandwidth Bandwidth parameter for the kernel
#'
#' @return A kernel matrix with the smile pattern
#'
#' @export
smile <- function(n, bandwidth = 2.0) {
  small <- ceiling(n^(1/2))
  eye_points <- small
  mouth_points <- ceiling(n/10)
  face_points <- n - 2 * eye_points - mouth_points
  
  X <- matrix(0, n, 2)
  idx <- 1
  
  # Eyes
  for (x_shift in c(-4.0, 4.0)) {
    for (i in 1:eye_points) {
      repeat {
        x <- 2 * runif(1) - 1
        y <- 2 * runif(1) - 1
        if (x^2 + y^2 <= 1.0) {
          X[idx, 1] <- x + x_shift
          X[idx, 2] <- y + 4.0
          idx <- idx + 1
          break
        }
      }
    }
  }
  
  # Mouth
  x_vals <- seq(-5.0, 5.0, length.out = mouth_points)
  for (x in x_vals) {
    X[idx, 1] <- x
    X[idx, 2] <- x^2 / 16.0 - 5.0
    idx <- idx + 1
  }
  
  # Face
  theta_vals <- seq(0, 2*pi, length.out = face_points)
  for (theta in theta_vals) {
    X[idx, 1] <- 10.0 * cos(theta)
    X[idx, 2] <- 10.0 * sin(theta)
    idx <- idx + 1
  }
  
  # Compute Gaussian kernel matrix
  D <- as.matrix(dist(X))
  A <- exp(-D^2/(2*bandwidth^2))
  
  return(A)
}

#ifndef RPCHOLESKY_H
#define RPCHOLESKY_H

// [[Rcpp::depends(RcppArmadillo)]]
#include <RcppArmadillo.h>

// Define R's random number generation for use in C++ code
#define R_NO_REMAP
#include <R.h>
#include <Rmath.h>

namespace rpcholesky {

/**
 * Rejection pivoted Cholesky decomposition
 *
 * @param H A positive definite matrix
 * @return A list containing the L matrix and the selected indices
 */
inline Rcpp::List rejection_cholesky(arma::mat& H) {
  int b = H.n_rows;
  if (H.n_rows != H.n_cols) {
    Rcpp::stop("rejection_cholesky requires a square matrix");
  }

  double trace = arma::trace(H);
  if (trace <= 0) {
    Rcpp::stop("rejection_cholesky requires a strictly positive trace");
  }

  // Pre-allocate vectors instead of creating them inside the loop
  arma::vec u(b);
  for (int j = 0; j < b; j++) {
    u(j) = H(j, j);
  }

  std::vector<int> idx;
  idx.reserve(b); // Pre-allocate for efficiency
  arma::mat L = arma::zeros(b, b);

  GetRNGstate(); // Get R's RNG state

  // Add maximum attempts limit to prevent infinite loops
  const int MAX_ATTEMPTS_PER_POSITION = 1000;

  for (int j = 0; j < b; j++) {
    int attempts = 0;
    bool accepted = false;

    // Limit number of attempts per position
    while (!accepted && attempts < MAX_ATTEMPTS_PER_POSITION) {
      attempts++;
      // Use R's uniform random number generator
      if (unif_rand() * u(j) < H(j, j)) {
        accepted = true;
        idx.push_back(j);
        L(arma::span(j, b-1), j) = H(arma::span(j, b-1), j) / std::sqrt(H(j, j));

        // Update the remaining submatrix
        for (int k = j+1; k < b; k++) {
          for (int l = j+1; l < b; l++) {
            H(k, l) -= L(k, j) * L(l, j);
          }
        }
      }
    }

    if (attempts >= MAX_ATTEMPTS_PER_POSITION && !accepted) {
      Rcpp::warning("Maximum rejection sampling attempts reached at position %d", j);
    }
  }

  PutRNGstate(); // Update R's RNG state

  // Create small L matrix with selected indices only
  arma::uvec accepted = arma::conv_to<arma::uvec>::from(idx);
  arma::mat L_small;
  if (accepted.n_elem > 0) {
    L_small = L.submat(accepted, accepted);
  } else {
    L_small = arma::zeros(0, 0);
  }

  // Return both the L matrix and the selected indices
  // Note: adding 1 to indices for R's 1-based indexing
  return Rcpp::List::create(
    Rcpp::Named("L") = L_small,
    Rcpp::Named("idx") = accepted + 1
  );
}

/**
 * Accelerated Randomly Pivoted Cholesky decomposition
 *
 * @param A A matrix
 * @param k The target rank of the approximation
 * @param b_in The block size (optional, default = k/10)
 * @param stoptol Tolerance for early stopping (default = 1e-13)
 * @param verbose Whether to print progress (default = false)
 * @return A list containing the factors and indices
 */
inline Rcpp::List accelerated_rpcholesky(arma::mat& A, int k,
                                         Rcpp::Nullable<int> b_in = R_NilValue,
                                         double stoptol = 1e-13,
                                         bool verbose = false) {
  int n = A.n_rows;
  int b;
  bool auto_b;

  // Handle the auto block size case
  if (b_in.isNull()) {
    b = std::ceil(k / 10.0);
    auto_b = true;
  } else {
    b = Rcpp::as<int>(b_in);
    auto_b = false;
  }

  // Validate inputs to prevent issues
  if (b <= 0) b = 1;
  if (k <= 0) {
    Rcpp::stop("Target rank k must be positive");
  }
  if (b > n) b = n;
  if (k > n) k = n;

  // Get diagonal elements
  arma::vec diags = A.diag();
  double orig_trace = arma::sum(diags);

  // Initialize result matrices - limit sizes for large matrices
  int actual_k = std::min(k, n); // Ensure we don't exceed matrix dimensions
  arma::mat G = arma::zeros(actual_k, n);
  arma::mat rows = arma::zeros(actual_k, n);
  arma::uvec arr_idx = arma::zeros<arma::uvec>(actual_k);

  int counter = 0;

  GetRNGstate(); // Get R's RNG state

  // Pre-allocate these outside the loop to avoid stack overflow
  arma::uvec idx(b);
  arma::mat H;
  arma::mat G_prev;
  arma::mat G_prev_subidx;
  Rcpp::List chol_result;
  arma::mat L;
  arma::uvec accepted;
  arma::uvec selected_idx;
  arma::mat G_new;

  // Use batch processing for very large matrices
  const int MAX_BATCH_SIZE = 10000; // Set a reasonable batch size

  while (counter < actual_k) {
    // Choose indices based on diagonal probabilities
    arma::vec prob = diags / arma::sum(diags);

    // Sample indices with replacement using efficient method
    for (int i = 0; i < b; i++) {
      double u = unif_rand();
      double cumsum = 0.0;
      for (int j = 0; j < n; j++) {
        cumsum += prob(j);
        if (u <= cumsum) {
          idx(i) = j;
          break;
        }
      }
    }

    // Timing for adaptive block size (if enabled)
    clock_t start_time = 0;
    clock_t middle_time = 0;
    clock_t end_time = 0;
    double rejection_time = 0.0;
    double process_time = 0.0;

    if (auto_b) {
      start_time = clock();
    }

    // Extract submatrices
    H = A.submat(idx, idx);

    // Subtract previous approximation
    if (counter > 0) {
      // Create matrices for the submatrix multiplication
      G_prev = G.rows(0, counter-1);
      G_prev_subidx.set_size(G_prev.n_rows, idx.n_elem);

      // Extract columns for idx
      for (size_t i = 0; i < idx.n_elem; i++) {
        G_prev_subidx.col(i) = G_prev.col(idx(i));
      }

      H -= G_prev_subidx.t() * G_prev_subidx;
    }

    // Add regularization to prevent numerical issues
    double eps = 1e-14 * arma::trace(H);
    if (eps > 0) {
      H.diag() += eps;
    }

    // Perform rejection sampling Cholesky with timeout
    try {
      chol_result = rejection_cholesky(H);
      L = Rcpp::as<arma::mat>(chol_result["L"]);
      accepted = Rcpp::as<arma::uvec>(chol_result["idx"]) - 1; // Convert back to 0-based
    } catch (std::exception &e) {
      if (verbose) {
        Rcpp::Rcout << "Warning: Cholesky decomposition failed. Adding regularization." << std::endl;
      }
      // Add stronger regularization and try again
      H.diag() += 1e-10 * arma::trace(H);
      chol_result = rejection_cholesky(H);
      L = Rcpp::as<arma::mat>(chol_result["L"]);
      accepted = Rcpp::as<arma::uvec>(chol_result["idx"]) - 1; // Convert back to 0-based
    }

    int num_sel = accepted.n_elem;

    if (num_sel > actual_k - counter) {
      num_sel = actual_k - counter;
      accepted = accepted.head(num_sel);
      L = L.submat(0, 0, num_sel-1, num_sel-1);
    }

    selected_idx.set_size(num_sel);
    for (int i = 0; i < num_sel; i++) {
      selected_idx(i) = idx(accepted(i));
    }

    if (auto_b) {
      middle_time = clock();
      rejection_time = ((double)(middle_time - start_time)) / CLOCKS_PER_SEC;
    }

    // Process selected indices in batches for large matrices
    for (int i = 0; i < num_sel; i++) {
      arr_idx(counter + i) = selected_idx(i);
    }

    // Extract rows for selected indices
    for (int i = 0; i < num_sel; i++) {
      rows.row(counter + i) = A.row(selected_idx(i));
    }

    // Compute residual in batches if needed
    G_new = rows.rows(counter, counter + num_sel - 1);
    if (counter > 0) {
      // Create matrices for the submatrix multiplication
      G_prev = G.rows(0, counter-1);

      // Process in batches if the matrix is very large
      if (selected_idx.n_elem > MAX_BATCH_SIZE) {
        G_prev_subidx.zeros(G_prev.n_rows, selected_idx.n_elem);

        for (size_t batch_start = 0; batch_start < selected_idx.n_elem; batch_start += MAX_BATCH_SIZE) {
          size_t batch_end = std::min(batch_start + MAX_BATCH_SIZE, (size_t)selected_idx.n_elem);
          size_t batch_size = batch_end - batch_start;

          arma::uvec batch_idx = selected_idx.subvec(batch_start, batch_end-1);

          for (size_t i = 0; i < batch_size; i++) {
            G_prev_subidx.col(batch_start + i) = G_prev.col(batch_idx(i));
          }
        }
      } else {
        G_prev_subidx.set_size(G_prev.n_rows, selected_idx.n_elem);
        // Extract columns for selected_idx
        for (size_t i = 0; i < selected_idx.n_elem; i++) {
          G_prev_subidx.col(i) = G_prev.col(selected_idx(i));
        }
      }

      G_new -= G_prev_subidx.t() * G_prev;
    }

    // Solve triangular system
    if (L.n_rows > 0) {
      try {
        G.rows(counter, counter + num_sel - 1) = arma::solve(trimatl(L), G_new);
      } catch (std::exception &e) {
        if (verbose) {
          Rcpp::Rcout << "Warning: Triangular solve failed. Using pseudoinverse." << std::endl;
        }
        // Use more stable but slower pseudoinverse if triangular solve fails
        G.rows(counter, counter + num_sel - 1) = arma::pinv(trimatl(L)) * G_new;
      }
    }

    // Update diagonal residuals
    for (int j = 0; j < n; j++) {
      double sum_sq = 0.0;
      for (int i = counter; i < counter + num_sel; i++) {
        sum_sq += G(i, j) * G(i, j);
      }
      diags(j) -= sum_sq;
      // Clip to avoid negative values due to numerical errors
      if (diags(j) < 0) diags(j) = 0;
    }

    if (auto_b) {
      end_time = clock();
      process_time = ((double)(end_time - middle_time)) / CLOCKS_PER_SEC;

      // Adjust block size adaptively
      // Aiming for rejection_time = process_time / 4
      int target = std::ceil(b * process_time / (4.0 * rejection_time));
      b = std::max({std::min({target, static_cast<int>(std::ceil(1.5 * b)), static_cast<int>(std::ceil(k / 3.0))}),
                   static_cast<int>(std::ceil(b / 3.0)), 10});
    }

    counter += num_sel;

    if (verbose && num_sel > 0) {
      Rcpp::Rcout << "Accepted " << num_sel << " / " << b << std::endl;
    }

    // Check early stopping criterion
    if (stoptol > 0 && arma::sum(diags) <= stoptol * orig_trace) {
      if (verbose) {
        Rcpp::Rcout << "Early stopping at iteration " << counter << std::endl;
      }
      G = G.rows(0, counter-1);
      rows = rows.rows(0, counter-1);
      arr_idx = arr_idx.head(counter);
      break;
    }

    // Add a check to avoid infinite loops if no pivots are being accepted
    if (num_sel == 0) {
      if (verbose) {
        Rcpp::Rcout << "Warning: No pivots accepted in this iteration. Increasing block size." << std::endl;
      }
      b = std::min(2 * b, n); // Double the block size but don't exceed matrix size

      // To prevent endless loops
      if (b >= n/2) {
        Rcpp::warning("Unable to find acceptable pivots, returning current approximation");
        break;
      }
    }

    // Periodic garbage collection to free memory
    if (counter % 1000 == 0) {
      R_CheckUserInterrupt(); // Allow user interruption for long-running processes
      R_gc(); // Force garbage collection
    }
  }

  PutRNGstate(); // Update R's RNG state

  // Return G matrix and indices
  // Add 1 to indices for R's 1-based indexing
  return Rcpp::List::create(
    Rcpp::Named("G") = G,
    Rcpp::Named("idx") = arr_idx.head(counter) + 1,
    Rcpp::Named("rows") = rows.rows(0, counter-1)
  );
}

/**
 * Compute approximation error of a low-rank approximation
 *
 * @param A The original matrix
 * @param G The G factor from rpcholesky
 * @param relative Whether to return the relative error (default = false)
 * @return The approximation error or relative error
 */
inline double compute_approximation_error(arma::mat& A, arma::mat& G, bool relative = false) {
  double trace_A = arma::trace(A);
  double trace_lra = arma::accu(G % G); // Squared Frobenius norm of G

  double error = trace_A - trace_lra;

  if (relative) {
    error /= trace_A;
  }

  return error;
}

/**
 * Create a Nystrom extension from sample
 *
 * @param A The original matrix
 * @param idx The indices selected by rpcholesky
 * @return A list containing the components of the Nystrom extension
 */
inline Rcpp::List nystrom_extension(arma::mat& A, arma::uvec& idx) {
  // Convert to 0-based indexing
  idx = idx - 1;

  // Extract the rows and the core
  arma::mat rows = A.rows(idx);
  arma::mat core = rows.cols(idx);

  // Make the core symmetric (handle numerical issues)
  core = (core + core.t()) / 2.0;

  // Compute the Cholesky factor of the core
  arma::mat L;
  bool chol_success = arma::chol(L, core);

  if (!chol_success) {
    // Add a small regularization if Cholesky fails
    double eps = arma::trace(core) * std::numeric_limits<double>::epsilon();
    arma::mat reg = eps * arma::eye<arma::mat>(core.n_rows, core.n_rows);
    chol_success = arma::chol(L, core + reg);

    if (!chol_success) {
      Rcpp::stop("Cholesky decomposition failed even with regularization");
    }
  }

  // Solve the triangular system
  arma::mat G = arma::solve(trimatl(L.t()), rows);

  // Convert back to 1-based indexing for R
  idx = idx + 1;

  return Rcpp::List::create(
    Rcpp::Named("G") = G,
    Rcpp::Named("core") = core,
    Rcpp::Named("rows") = rows,
    Rcpp::Named("idx") = idx
  );
}

/**
 * Apply the Nystrom approximation to multiply with a matrix
 *
 * @param nystrom The Nystrom extension returned by nystrom_extension
 * @param B The matrix to multiply with
 * @return The product of the Nystrom approximation and B
 */
inline arma::mat nystrom_mult(Rcpp::List& nystrom, arma::mat& B) {
  arma::mat core = Rcpp::as<arma::mat>(nystrom["core"]);
  arma::mat rows = Rcpp::as<arma::mat>(nystrom["rows"]);

  // rows * B = X, then solve core * Y = X for Y, then rows^T * Y
  return rows.t() * arma::solve(core, rows * B);
}

#endif // RPCHOLESKY_H
